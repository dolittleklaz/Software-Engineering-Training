CREATE TABLE manufacturer (
	 ManufacturerID   INT  NOT NULL,
	 ManufacturerName   text  NOT NULL,
PRIMARY KEY 
(	 ManufacturerID ))


CREATE TABLE Distributor (
    DistributorID INT NOT NULL,
    DistributorName VARCHAR(255) NOT NULL,
	ManufacturerID INT NOT NULL,
    PRIMARY KEY (DistributorID, ManufacturerID),
	FOREIGN KEY (ManufacturerID) REFERENCES Manufacturer(ManufacturerID)
);
CREATE TABLE Distproduct (
'[+*
    DistributorID INT NOT NULL,
    ProductID INT NOT NULL,
    DistProductID INT NOT NULL,
    PRIMARY KEY (DistributorID, ProductID, DistProductID),
	FOREIGN KEY (ProductID) REFERENCES Product(ProductID)
);
CREATE TABLE Payment (
    PayID INT NOT NULL,
    Paytype VARCHAR(255) NOT NULL,
    PRIMARY KEY (PayID)
);
